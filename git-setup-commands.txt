# Git Repository Setup Commands
# Run these commands in Git Bash or PowerShell (restart PowerShell if git command not found)

# Navigate to your project
cd "c:\Resume Site"

# Configure Git (one-time setup - use your email)
git config --global user.name "Romone Dunlop"
git config --global user.email "romonedunlop2@gmail.com"

# Initialize Git repository
git init

# Add all files to staging
git add .

# Create initial commit
git commit -m "Initial commit - Complete resume website with testimonials and PDF"

# Check status (should say "nothing to commit, working tree clean")
git status
